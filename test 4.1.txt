  #include <stdio.h>

int isComposite(int num) {
    if (num <= 1) {
        return 0;
    }
    for (int i = 2; i < num; i++) {
        if (num % i == 0) {
            return 1;
        }
    }
    return 0;
}

int main() {
    int arr[] =   {26, 28, 37, 26, 33, 31, 29 };
    int n = sizeof(arr) / sizeof(arr[0]);
    int count = 0;

    for (int i = 0; i < n; i++) {
        if (isComposite(arr[i])) {
            count++;
        }
    }

    printf("Number of Composite Numbers = %d", count);

    return 0;
}
