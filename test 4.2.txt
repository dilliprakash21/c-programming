 #include <stdio.h>

int main() {
    int array[10] = {-26, 28, 37, -26, 33, -31, -29};
    int i, count = 0;

    for (i = 0; i < 10; i++) {
        if (array[i] < 0) {
            count++;
        }
    }

    printf("There are %d negative numbers in the array.\n", count);

    return 0;
}


out put:There are 4 negative numbers in the array.
